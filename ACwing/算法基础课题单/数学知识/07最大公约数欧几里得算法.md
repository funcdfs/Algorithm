辗转相除法

欧几里得算法

``` cpp 
#include <iostream>
#include <algorithm>

using namespace std;

int gcd(int a, int b) { 
    return b ? gcd(b, a % b) : a;
}

int main() {
    int n;
    cin >> n;
    while (n--) {
        int a, b;
        scanf("%d%d", &a, &b);
        printf("%d\n", gcd(a, b));
    }
    return 0;
}
```
